8   import com.edii.j211.logiikka.MuokattavaPelikentta;
9   import com.edii.j211.logiikka.Pelikentta;
10  import java.util.ArrayList;
11  import java.util.List;
12  
13  /**
14   *
15   * @author edii
16   */
17  public final class Util {    
18      public static void tyhjenna(MuokattavaPelikentta kentta) {
19          for (int y = 0; y < kentta.koko(); y++) {
20              for (int x = 0; x < kentta.koko(); x++) {
21                  kentta.asetaArvo(x, y, 0);
22              }
23          }
24      }
25      
26      public static int[][] tyhjat(Pelikentta kentta) {
27          List<int[]> out = new ArrayList<>();
28          
29          for (int y = 0; y < kentta.koko(); y++) {
30              for (int x = 0; x < kentta.koko(); x++) {
31                  if (kentta.arvo(x, y) == 0) {
32                      out.add(new int[] { x, y });
33                  }
34              }
35          }
36          
37          int[][] outArr = new int[out.size()][];
38          return out.toArray(outArr);
39      }    
40  }
