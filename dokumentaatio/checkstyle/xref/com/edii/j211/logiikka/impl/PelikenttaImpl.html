3   import com.edii.j211.logiikka.MuokattavaPelikentta;
4   
5   /**
6    * Pelikentta-rajapinnan toteutus.
7    *
8    * Toteuttaa laajemman MuokattavaPelikenttä rajapinnan, jota {@link PeliImpl}
9    * käyttää sisäisesti, mutta palauttaa muille näkyvänä Pelikentta-olion, jota ei
10   * voi muokata.
11   *
12   * @see com.edii.j211.logiikka.MuokattavaPelikentta
13   * @see com.edii.j211.logiikka.Pelikentta
14   * @see com.edii.j211.logiikka.impl.PeliImpl
15   */
16  public class PelikenttaImpl implements MuokattavaPelikentta {
17  
18      private int koko;
19      private int[][] kentta;
20  
21      public PelikenttaImpl(int koko) {
22          this.koko = koko;
23          this.kentta = new int[koko][koko];
24      }
25  
26      @Override
27      public int koko() {
28          return koko;
29      }
30  
31      @Override
32      public void asetaArvo(int x, int y, int arvo) {
33          this.kentta[y][x] = arvo;
34      }
35  
36      @Override
37      public int arvo(int x, int y) {
38          return this.kentta[y][x];
39      }
40  }
