4   import com.edii.j211.logiikka.Pelikentta;
5   import java.awt.Color;
6   import java.awt.Font;
7   import java.awt.GridLayout;
8   import java.util.HashMap;
9   import java.util.Map;
10  import javax.swing.JLabel;
11  import javax.swing.JPanel;
12  import static javax.swing.SwingConstants.CENTER;
13  
14  /**
15   * Pelialue, jolla graafinen käyttöliittymä renderöi pelin tilanteen.
16   *
17   * @see GraafinenKayttoliittyma
18   */
19  public class Pelialue extends JPanel {
20  
21      /**
22       * Peli, jota tämä Pelialue kuvastaa.
23       */
24      private Peli peli;
25  
26      /**
27       * Pelin jokainen ruutu näytetään omana JLabelina, tämä taulukko säilyttää
28       * jokaista ruutua edustavan JLabelin.
29       */
30      private JLabel[][] ruudut;
31  
32      /**
33       * Ainoa konstruktori pelialueen luomiseen. Saa parametrinaan pelin, jonka
34       * avulla määräytyy minkä kokoinen pelialue tulee näyttää ja mikä pelin
35       * tilanne on.
36       *
37       * @param peli Peli, joka esitetään tällä pelialueella
38       * @see com.edii.j211.logiikka.Peli
39       */
40      public Pelialue(Peli peli) {
41          super(new GridLayout(peli.koko(), peli.koko()));
42          this.peli = peli;
43  
44          ruudut = new JLabel[peli.koko()][peli.koko()];
45  
46          for (int y = 0; y < peli.koko(); y++) {
47              for (int x = 0; x < peli.koko(); x++) {
48                  ruudut[y][x] = new JLabel("");
49                  this.add(ruudut[y][x]);
50                  ruudut[y][x].setHorizontalAlignment(CENTER);
51                  ruudut[y][x].setVerticalAlignment(CENTER);
52                  ruudut[y][x].setBackground(Color.white);
53                  ruudut[y][x].setOpaque(true);
54              }
55          }
56      }
57  
58      /**
59       * Pitää sisällään eri pistearvojen näyttöön tarkoitetut edusvärit (numeron
60       * väri).
61       */
62      private static final Map<Integer, Integer> EDUSVARIT = new HashMap<>();
63  
64      /**
65       * Pitää sisällään eri pistearvojen näyttöön tarkoitetut taustavärit.
66       */
67      private static final Map<Integer, Integer> TAUSTAVARIT = new HashMap<>();
68  
69      static {
70          EDUSVARIT.put(2, 0x776e65);
71          EDUSVARIT.put(4, 0x776e65);
72          TAUSTAVARIT.put(0, 0xcdc1b4);
73          TAUSTAVARIT.put(2, 0xeee4da);
74          TAUSTAVARIT.put(4, 0xede0c8);
75          TAUSTAVARIT.put(8, 0xf2b179);
76          TAUSTAVARIT.put(16, 0xf59563);
77          TAUSTAVARIT.put(32, 0xf67c5f);
78          TAUSTAVARIT.put(64, 0xf65e3b);
79          TAUSTAVARIT.put(128, 0xedcf72);
80          TAUSTAVARIT.put(256, 0xedcc61);
81          TAUSTAVARIT.put(512, 0xedc850);
82          TAUSTAVARIT.put(1024, 0xedc53f);
83          TAUSTAVARIT.put(2048, 0xedc22e);
84      }
85  
86      /**
87       * Päivitä pelialue vastaamaan nykytilannetta.
88       */
89      public void paivita() {
90          Pelikentta kentta = peli.pelikentta();
91  
92          for (int y = 0; y < peli.koko(); y++) {
93              for (int x = 0; x < peli.koko(); x++) {
94                  int arvo = kentta.arvo(x, y);
95                  Color fg = new Color(EDUSVARIT.getOrDefault(arvo, 0xf9f6f2));
96                  Color bg = new Color(TAUSTAVARIT.getOrDefault(arvo, 0x000000));
97  
98                  int height = ruudut[y][x].getHeight();
99                  ruudut[y][x].setFont(new Font("Arial", Font.BOLD, height / 3));
100 
101                 if (arvo == 0) {
102                     ruudut[y][x].setText("");
103                 } else {
104                     ruudut[y][x].setText(Integer.toString(arvo));
105                 }
106 
107                 ruudut[y][x].setForeground(fg);
108                 ruudut[y][x].setBackground(bg);
109             }
110         }
111 
112         this.repaint();
113         this.revalidate();
114     }
115 }
